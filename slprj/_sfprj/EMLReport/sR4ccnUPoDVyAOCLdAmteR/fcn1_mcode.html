1function force = fcn(z,v)
 2%#codegen
 3
 4%% parameters
 5upbound = 2.5;
 6lobound = 0.0;
 7upstiff = 10e-6;
 8lostiff = 2e6;
 9updamp = 1.5e4;
10lodamp = 150;
11
12%% equations
13if (z > upbound)
14    force = -1 * (upstiff * (z - upbound) + updamp * v);
15elseif (z < lobound)
16    force = abs(lostiff * (z - lobound) + lodamp * v);
17else
18    force = 0;
19end
